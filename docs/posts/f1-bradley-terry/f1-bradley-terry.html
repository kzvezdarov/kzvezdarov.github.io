def rank_order_row(df, row):
    losers = df[df["finishing_pos"] > row["finishing_pos"]][["driver", "constructor"]]
    return pd.Series(
        [
            row["driver"],
            row["constructor"],
            losers["driver"],
            losers["constructor"]
        ],
        index=["winner", "winner_constructor", "loser", "loser_constructor"]
    )
    
def explode_rank_order(df):
    return df.apply(lambda row: rank_order_row(df, row), axis=1).explode(["loser", "loser_constructor"])
